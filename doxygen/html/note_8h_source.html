805// The Notecard, whose default I2C address is below, uses a serial-to-i2c
806// protocol whose "byte count" must fit into a single byte and which must not
807// include a 2-byte header field. This is why the maximum that can be
808// transmitted by note-c in a single I2C I/O is 255 - 2 = 253 bytes.
809
813#define NOTE_I2C_ADDR_DEFAULT 0x17
814
815// Serial-to-i2c protocol header size in bytes
816#ifndef NOTE_I2C_HEADER_SIZE
817# define NOTE_I2C_HEADER_SIZE 2
818#endif
819
820// Maximum bytes capable of being transmitted in a single read/write operation
821#ifdef NOTE_I2C_MAX_MAX
822# ifdef NOTE_I2C_MTU_MAX
823# undef NOTE_I2C_MTU_MAX
824# endif
825#else
826# ifdef NOTE_I2C_MTU_MAX
827# define NOTE_I2C_MAX_MAX NOTE_I2C_MTU_MAX
828# else
829# define NOTE_I2C_MAX_MAX (UCHAR_MAX - NOTE_I2C_HEADER_SIZE)
830# endif
831#endif
832#ifndef NOTE_I2C_MTU_MAX
833# define NOTE_I2C_MTU_MAX NOTE_I2C_MAX_MAX
834#endif

836// In ARDUINO implementations, which to date is the largest use of this library,
837// the Wire package is implemented in a broad variety of ways by different
838// vendors. The default implementation has a mere 32-byte static I2C buffer,
839// which means that the maximum to be transmitted in a single I/O (given our
840// 2-byte serial-to-i2c protocol header) is 30 bytes. However, if we know
841// the specific platform (such as STM32DUINO) we can relax this restriction.
842#ifdef NOTE_I2C_MAX_DEFAULT
843// user is overriding it at compile time
844# ifdef NOTE_I2C_MTU_DEFAULT
845# undef NOTE_I2C_MTU_DEFAULT
846# endif
847#else
848# if defined(NOTE_I2C_MTU_DEFAULT)
849// user is overriding it at compile time
850# define NOTE_I2C_MAX_DEFAULT NOTE_I2C_MTU_DEFAULT
851# elif defined(ARDUINO_ARCH_STM32)
852// we know that stm32duino dynamically allocates I/O buffer
853# define NOTE_I2C_MAX_DEFAULT NOTE_I2C_MTU_MAX
854# else
855// default to what's known to be safe for all Arduino implementations
860# define NOTE_I2C_MAX_DEFAULT 30
861# endif
862#endif
863#ifndef NOTE_I2C_MTU_DEFAULT
864# define NOTE_I2C_MTU_DEFAULT NOTE_I2C_MAX_DEFAULT
865#endif
