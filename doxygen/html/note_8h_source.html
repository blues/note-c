804// The Notecard, whose default I2C address is below, uses a serial-to-i2c
805// protocol whose "byte count" must fit into a single byte and which must not
806// include a 2-byte header field. This is why the maximum that can be
807// transmitted by note-c in a single I2C I/O is 255 - 2 = 253 bytes.
808
812#define NOTE_I2C_ADDR_DEFAULT 0x17
813
814// Serial-to-i2c protocol header size in bytes
815#ifndef NOTE_I2C_HEADER_SIZE
816# define NOTE_I2C_HEADER_SIZE 2
817#endif

819// Maximum bytes capable of being transmitted in a single read/write operation
820#ifdef NOTE_I2C_MAX_MAX
821# ifdef NOTE_I2C_MTU_MAX
822# undef NOTE_I2C_MTU_MAX
823# endif
824#else
825# ifdef NOTE_I2C_MTU_MAX
826# define NOTE_I2C_MAX_MAX NOTE_I2C_MTU_MAX
827# else
828# define NOTE_I2C_MAX_MAX (UCHAR_MAX - NOTE_I2C_HEADER_SIZE)
829# endif
830#endif
831#ifndef NOTE_I2C_MTU_MAX
832# define NOTE_I2C_MTU_MAX NOTE_I2C_MAX_MAX
833#endif

835// In ARDUINO implementations, which to date is the largest use of this library,
836// the Wire package is implemented in a broad variety of ways by different
837// vendors. The default implementation has a mere 32-byte static I2C buffer,
838// which means that the maximum to be transmitted in a single I/O (given our
839// 2-byte serial-to-i2c protocol header) is 30 bytes. However, if we know
840// the specific platform (such as STM32DUINO) we can relax this restriction.
841#ifdef NOTE_I2C_MAX_DEFAULT
842// user is overriding it at compile time
843# ifdef NOTE_I2C_MTU_DEFAULT
844# undef NOTE_I2C_MTU_DEFAULT
845# endif
846#else
847# if defined(NOTE_I2C_MTU_DEFAULT)
848// user is overriding it at compile time
849# define NOTE_I2C_MAX_DEFAULT NOTE_I2C_MTU_DEFAULT
850# elif defined(ARDUINO_ARCH_STM32)
851// we know that stm32duino dynamically allocates I/O buffer
852# define NOTE_I2C_MAX_DEFAULT NOTE_I2C_MTU_MAX
853# else
854// default to what's known to be safe for all Arduino implementations
859# define NOTE_I2C_MAX_DEFAULT 30
860# endif
861#endif
862#ifndef NOTE_I2C_MTU_DEFAULT
863# define NOTE_I2C_MTU_DEFAULT NOTE_I2C_MAX_DEFAULT
864#endif
